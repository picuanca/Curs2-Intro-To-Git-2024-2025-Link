## initializeaza proiectul de git local - creaza un folder .git ascuns
git init

## Descarca de pe server un proiect de git
git clone https://github.com/username/project

## Arata lista de commituri
git log
git log --oneline

## Arata statusul fisierelor (daca sunt schimbate)
git statusul
git status -schimbate


### Comenzi de adaugare a fisierelor in STAGED (inainte de crearea a comitului)
git add filename
git add .
git add filename.*
git add filename1 filename2 filename3

## Setarea numelui global

# Setarea numelui utilizatorului
git config --global user.name "Numele tau..."
# Citirea numelui utilizatorului
git config --global user.nume

# Setarea e-mailului utilizatorului
git config --global user.email "emailul@tau.com"
# Citirea e-mailului utilizatorului
git config --global user.email

### Comenzi de creare a commitului
git commit -m " Acesta este un mesaj descriptiv al commitului "
git commit --message " Acesta este un mesaj descriptiv al commitului "

### Comanda de adauga + commit (!!! Fisierele trebuie sa fi fost incluse intr-un commit inainte)
git commit -am " Acesta este un mesaj descriptiv "
